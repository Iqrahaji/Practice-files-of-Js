=> index number: [0] , [1] , [2]
console.log(fruits[1]); => Output: Banana

fruits[1] = "Orange"; console.log(fruits); => Output: ["Apple", "Orange", "Mango"]

fruits.push("Pineapple"); console.log(fruits); => Output: ["Apple", "Orange", "Mango", "Pineapple"]

fruits.pop(); console.log(fruits); => Output: ["Apple", "Orange", "Mango"]

fruits.shift(); console.log(fruits); => Output: ["Orange", "Mango"]

fruits.unshift("Strawberry"); console.log(fruits); => Output: ["Strawberry", "Orange", "Mango"]

console.log(fruits.length); => Output: 4

Object

An object is a collection of properties, and a property is an association between a name (or key) and a value.





//changing the age property

//adding a new property to the object

//removing a property from the object 		let person = {
firstName: "John",
lastName: "Doe",
age: 40
};
console.log(person.firstName); => Output: John

person.age = 41; console.log(person.age); // Output: 41

person.city = "New York"; console.log(person.city); // Output: New York

delete person.lastName; console.log(person); // Output: {firstName: "John", age: 41, city: "New York"}
